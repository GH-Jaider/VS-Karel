BEGINNING-OF-PROGRAM
	DEFINE-NEW-INSTRUCTION turnright AS
	BEGIN
		turnleft;
		turnleft;
		turnleft
	END
	DEFINE-NEW-INSTRUCTION beeper-line AS
	BEGIN
		ITERATE 5 TIMES
		BEGIN
			putbeeper;
			move
		END
	END
	DEFINE-NEW-INSTRUCTION picking AS
	BEGIN
		WHILE next-to-a-beeper DO
		BEGIN
			pickbeeper;
			move
		END
	END
	BEGINNING-OF-EXECUTION
		move;
		beeper-line;
		turnright;
		turnright;
		move;
		picking;
		turnoff
	END-OF-EXECUTION
END-OF-PROGRAM
